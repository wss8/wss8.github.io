1.1、auto、decltype

auto的基本使用

  • 在C++11之前，auto关键字是栈变量的默认修饰符
  • C++11 使用 auto 和 decltype 这两个关键字实现类型推导，让编译器来推导变量的类型
  • C++11中auto是一种类型占位符，编译器可以从初始化式中推断出变量的实际类型(使用模板参数推导规则推导类型）
    • auto a = 10; auto b = a; auto c = 1.0; auto* p = &a; auto& r = a;
  • auto 定义时变量时必须初始化（保证了类型安全），auto不能定义数组，不能作为模板参数。
    • auto a; auto c[2] = {0}; vector<auto> vec;
  • 在类中，auto必须配合static定义变量，且必须初始化
    • static const auto i = 10 （有初始值的静态数据成员必须为const）
  • C++11 auto不允许用作函数参数与返回值类型，C++14才允许auto用于函数返回值并会被推导，C++14的lambda函数允许在形参中使用auto
    • auto func() {  return 0; }
    • auto f = [](auto a, auto b){return a + b;};
  • auto推导规则
    • 使用auto定义的变量为指针或引用时，会保留右操作数的cv属性（const、volatile）
      • const auto a = 10; auto& b = a; // b为const int&
    • 使用auto定义的变量为非指针或非引用时，会忽略右操作数的cv属性
      • const auto a = 10; auto b = a; // b为int

decltype 的基本使用

  • decltype ( entity ) 、 decltype ( expression )
    • 从表达式的类型或变量的类型推断出要定义的变量的类型（关于表达式的两种属性：类型和值类别前文已有介绍）
    • int x = 10; decltype(x) y1;
  • decltype总是不加修改的产生变量或者表达式的类型，即保留表达式的cv属性与引用
    • cont int &a= 1; decltype(a) b = 1; // b类型：const int&
  • decltype推导不同表达式会有不同的类型
    • 当表达式值类别为 lvalue表达式，decltype会推出T的引用，即T&
      • int x = 10; decltype((x)) y2; // (x)是lvalue表达式，y2类型为int&，必须初始化，所以这里会报错
    • 当表达式值类别为 prvalue表达式，decltype会推出T的引用，即T
      • int a = 10; int b = 10; decltype(a + b) y1; decltype(a+=b) y2 = y1;// 类型分别为int和int&
    • 当表达式值类别为 xvalue表达式，decltype会推出T的引用，即T&&
      • decltype(std::move(Test())) x = Test(); // std::move(Test())是xvalue表达式，所以x类型为Test&&
  • 尾置返回类型（trailing return type）
    • 参数列表后面使用 -> 指出函数的返回类型， 参数前使用auto仅仅占位，没有功能。为什么要后置？因为decltype(x) 如果放前面，参数还没定义
#include <iostream>
#include <functional>
#include <utility>
using namespace std;

int test() {
    cout << "test()" << endl;
    return 0;
}
int main()
{
    const int cx = 10;
    const int& crx = cx;
    const int* const cpx = &cx;

    auto x1 = cx;         // x1: int           
    auto& x2 = cx;        // x2: const int ，保留cv属性     
    const auto& x3 = cx;  // x3: const int&, 保留cv属性   
    auto x4 = crx;        // x4：int           
    auto x5 = cpx;        // x5：const int*，*cpx的const属性会保留，而修饰cpx的const被忽略              

    static_assert(is_same<decltype(x1), int>::value, "");        // ok
    static_assert(is_same<decltype(x2), const int&>::value, ""); // ok
    static_assert(is_same<decltype(x3), const int&>::value, ""); // ok
    static_assert(is_same<decltype(x4), int>::value, "");        // ok
    static_assert(is_same<decltype(x5), const int*>::value, ""); // ok

    decltype(cx) d_cx = 20;   // const int
    decltype(crx) d_crx = 20; // const int&

    int i = 10;
    int* p = &i;
    decltype(i) i1 = 0;       // int   
    decltype(*p) i2 = i;      // int&  lvalue表达式
    decltype((i)) i3 = i;     // int&, lvalue表达式
    decltype(i = 20) i4 = i;  // int&, lvalue表达式，不会执行 i = 20，
    decltype(i3 + 7) i5 = 0;  // int , rvalue表达式

    decltype(test()) i6 = 0;  // int , lvalue表达式, 编译期间推断，不会执行函数
    decltype(test) f1;        // int (void), 函数类型，不是函数指针
    static_assert(is_same<decltype(f1), int(void)>::value, ""); // ok
    function<decltype(test)> f2 = test; //  可调用对象
    f2();

    return 0;
}

// C++ 11之前的写法
template <typename T1, typename T2, typename T3>
T1 Add(T2 a, T3 b){
    return static_cast<T1>(a + b);
}
// C++ 11之后的写法
template <typename T, typename Y>
decltype(v1 + v2) add(T v1, Y v2) {
    return v1 + v2;
}
template <typename T, typename U>
decltype((*(T*)0) + (*(U*)0)) add(T t, U u) {
    return t + u; // decltype(expr) 表达式并不会执行，所以不会异常
}
template <typename T, typename U>
auto add(T t, U u) -> decltype(t + u) {
    return t + u;
}
// c++14 支持下面写法
template <typename T, typename U>
decltype(auto) add(T t, U u) {
    return t + u;
}

为什么要使用 decltype(auto)?

// C++14
auto beTen(int& i)
{
    i = 10;
    return i;
    // 相当于 auto ret = i;    期望auto被推导为int&, 但实际为int 
    // 使用auto&可以解决，auto& ret = i; 但不够便捷
    // 使用decltype(auto)可以原封不动返回i的类型(auto代表返回值会要类型推导，decltype代表不加修改)
}
int main()
{
    int i = 0;
    //beTen(i) = 20;

    decltype(auto) a = i;

    return 0;
}
// decltype(auto) 大大较少代码量

1.2、模板类型推导

在C++11中可以将模板参数分为下面几种情况

// T parm： 模板参数类型是T
template <typename T>
void func_r(T parm);

// T& parm： 模板参数类型是一个引用
template <typename T>
void func_r(T& parm);

// T&& parm： 模板参数类型是一个通用引用
template <typename T>
void func_rr(T&& parm);

模板类型推导时，不论模板参数如何，所有有引用的实参会被视为无引用
当模板参数类型是一个T类型时

  • 此时为值拷贝传参， 实参入栈，形参在自己的栈帧里拷贝实参内容到自己
  • 如果实参有 const 或 volatile属性(简称cv)，都会被忽略

当模板参数类型是一个引用或指针时

  • 实参入栈，形参为指针或引用作为新栈帧的一个局部变量，占4字节或8字节，指向实参（ebp+8）
  • 如果实参有 const 或 volatile属性，都会被保留

当模板参数类型是一个通用引用时

  • 传入参数是lvalue，被推导为左值引用，传入参数为rvalue被推导为右值引用
#include <iostream>
#include <boost/type_index.hpp>

using namespace std;
using boost::typeindex::type_id_with_cvr;  // Effective Modern C++有介绍：可以查看编译器的模板推导类型结果


// 注意：模板类型推导时，不论模板参数如何，所有有引用的实参会被视为无引用

// T parm： 模板参数类型是T, 传值
template <typename T>
void func_v(T parm)
{
    cout << parm << endl;
    cout << "T: " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl;
}
// T parm： 两个参数模板参数类型是T,int 传值
template <typename T, int N>
int func_arr(T(&parm)[N])
{
    cout << "N = " << N << endl;
    cout << "T: " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl;
    return N;
}

// T& parm： 模板参数类型是一个引用或指针时
template <typename T>
void func_r(T& parm)
{
    cout << "T: " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl;
}
// T&& parm： 模板参数类型是一个通用引用
template <typename T>
void func_rr(T&& parm)
{
    cout << "T: " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl;
}

void test(void) {}

int main()
{
    int x = 10;
    const int cx = x; 
    const int& crx = cx;
    const char* str1 = "const char* str1";
    char* const str2 = "char* const str2";
    const char arr[] = "const char arr[]";

                  // 根据实参与给定模板推测模板T的类型
                  // T 类型              实参
    func_v(10);   // T ->  int             [无变动]
    func_v(x);    // T ->  int             [无变动]
    func_v(cx);   // T ->  int            [忽略const]
    func_v(crx);  // T ->  int            [忽略const &]
    func_v("1");  // T -> const char*   [const修饰的是内容，不是指针变量，所以不忽略]
    func_v(str1); // T -> const char*   
    func_v(str2); // T -> char*         [忽略const]
    func_v(arr);  // T -> const char*   [编译时退化为指针]
    func_arr(arr);// T -> const char  N = 17  
    func_v(test); // T -> void (__cdecl*)(void)

    cout << endl;

    func_r(x);    // T = int        parm : int & 
    func_r(cx);   // T = int const  parm : int const &    [保留const]
    func_r(crx);  // T = int const  parm : int const &    [忽略&]
    func_r(arr);  // T -> const char (&) [17]              
    func_r(test); // parm: void (__cdecl&)(void)         

    cout << endl;

    func_rr(x);   // x是lvalue， T -> int&         parm : int& && -> int&
    func_rr(cx);  // cx是lvalue，T -> const int&   parm : const int& && -> const int&
    func_rr(crx); // rx是lvalue，T -> const int&   parm : const int& && -> const int&
    func_rr(10);  // 10是rvalue，T -> int          parm : int&& 
    func_rr(std::move(x));//     T -> int          parm : int&& 

    return 0;
}

1.3、auto类型推导原理

#include <iostream>
#include <vector>
#include <boost/type_index.hpp>

using namespace std;
using boost::typeindex::type_id_with_cvr;

// T parm 
template <typename T>
void func_v(T parm)
{
    cout << "auto == T == " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl << endl;
}
// const T parm 
template <typename T>
void func_cv(const T parm)
{
    cout << "auto == T == " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl << endl;
}
// T& parm 
template <typename T>
void func_rv(T& parm)
{
    cout << "auto == T == " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl << endl;
}
// const T& parm 
template <typename T>
void func_crv(const T& parm)
{
    cout << "auto == T == " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl << endl;
}

template <typename T>
void func_cp(const T* parm)
{
    cout << "auto == T == " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl << endl;
}
template <typename T>
void func_pc(T* const parm)
{
    cout << "auto == T == " << type_id_with_cvr<T>().pretty_name() << endl;
    cout << "ParmType: " << type_id_with_cvr<decltype(parm)>().pretty_name() << endl << endl;
}

void test(int, int) {}

int main()
{   
    // 把auto替换为T就是要匹配的模板类型， 右操作数被当作实参，左侧部分被当作模板函数
    auto x = 10; // 匹配 T parm     
    func_v(10);  // T -> int    ParmType: int    

    const auto cx = x; // 匹配 const T parm    
    func_cv(x);        // T == int

    auto& rcx = cx;   
    func_rv(cx);      //  T == int const 

    const auto& crx = cx;  
    func_rv(rcx); //  T == int const
    auto x2 = cx;  
    func_v(cx);   // T == int
    auto x3 = crx;         
    func_v(crx);  // T == int
    auto x4 = new int(100); 
    func_v(new int(100)); // T == int *      

    cout << "1-----" << endl;

    const char arr[] = "wss";
    auto arr1 = arr;       
    auto& arr2 = arr;    
    func_v(arr);  // T == char const *     ParmType: char const *
    func_rv(arr); // T == char const [4]   ParmType: char const (&)[4]

    cout << "2-----" << endl;

    auto f1 = test;    // 匹配模板：T parm
    auto& f2 = test;   // 匹配模板：T& parm
    func_v(test);    //  T == void (__cdecl*)(int,int) ParmType: void (__cdecl*)(int,int)
    func_rv(test);   //  T == void __cdecl(int,int)    ParmType: void (__cdecl&)(int,int)

    cout << "3-----" << endl;

    auto str0 = "00";
    const auto str1 = "1";   // 匹配模板：const T parm
    const auto* str2 = "2";  // 匹配模板：const T* parm
    auto* const str3 = "3";  // 匹配模板：T* const parm

    func_v("00"); 
    func_cv("1"); 
    func_cp("2"); 
    func_pc("3");

    int n1 = { 0 };   // int n1{ 0 };
    auto n2 = { 0 };  // 右操作数发生了隐式类型转换
    auto n3{0};       // 和上面区别一下

    func_v(n2); // T == class std::initializer_list<int>   ParmType: class std::initializer_list<int>
    func_v(n3); // T == int  ParmType: int

    return 0;
}
